cd ~/skillhub-app

# 1. Remove incorrect file/directory if it exists
sudo rm -rf prometheus/prometheus.yml

# 2. Ensure directory exists
mkdir -p prometheus

# 3. Create the config file
cat > prometheus/prometheus.yml << 'EOF'
global:
  scrape_interval: 15s
  evaluation_interval: 15s

scrape_configs:
  - job_name: 'skillhub-backend'
    metrics_path: '/actuator/prometheus'
    static_configs:
      - targets: ['backend:8080']
        labels:
          service: 'skillhub-backend'
          
  - job_name: 'prometheus'
    static_configs:
      - targets: ['localhost:9090']
EOF

# 4. Verify file was created correctly
ls -la prometheus/prometheus.yml
cat prometheus/prometheus.yml

# 5. Start Prometheus
sudo docker-compose up -d prometheus

# 6. Wait a moment
sleep 5

# 7. Check logs (should have output now!)
sudo docker logs skillhub-prometheus --tail 30

# 8. Test Prometheus
curl http://localhost:9091
